class HeartbeatEngine:
    """心跳引擎 - 支持AI主动发言"""
    
    def __init__(self, event_bus: IEventBus):
        self.event_bus = event_bus
        self.timers: Dict[str, asyncio.TimerHandle] = {}
    
    async def start(self, session_id: str):
        """启动心跳"""
        loop = asyncio.get_event_loop()
        
        async def tick():
            await self.event_bus.publish(Event(
                type=EventType.AI_HEARTBEAT,
                source="heartbeat",
                session_id=session_id,
                data={"timestamp": datetime.now().isoformat()}
            ))
            # 调度下一次
            self.timers[session_id] = loop.call_later(
                5.0,  # 5秒心跳
                lambda: asyncio.create_task(tick())
            )
        
        await tick()
    
    async def stop(self, session_id: str):
        """停止心跳"""
        if session_id in self.timers:
            self.timers[session_id].cancel()
            del self.timers[session_id]
```

class PluginManager:
    """插件管理器 - 支持热插拔"""
    
    def __init__(self, plugin_dir: Path):
        self.plugin_dir = plugin_dir
        self.plugins: Dict[str, Plugin] = {}
    
    async def load_plugins(self):
        """加载插件"""
        for plugin_path in self.plugin_dir.glob("*.py"):
            if plugin_path.name.startswith("_"):
                continue
            
            module = importlib.import_module(f"plugins.{plugin_path.stem}")
            
            if hasattr(module, "plugin"):
                plugin = module.plugin
                await plugin.on_load()
                self.plugins[plugin.id] = plugin
    
    async def unload_plugin(self, plugin_id: str):
        """卸载插件"""
        if plugin_id in self.plugins:
            await self.plugins[plugin_id].on_unload()
            del self.plugins[plugin_id]

# 插件接口
class Plugin(ABC):
    @abstractmethod
    def id(self) -> str: ...
    
    @abstractmethod
    async def on_load(self): ...
    
    @abstractmethod
    async def on_unload(self): ...
```
