# 规则模块设计

## 一、模块职责

RuleModule 负责管理讨论规则的定义、验证和执行。核心功能包括：

1. **规则模板管理**：创建、读取、更新、删除规则模板
2. **规则实例化**：根据模板创建运行时规则
3. **约束评估**：评估消息和行为是否满足约束条件
4. **触发器管理**：检测触发条件并执行相应动作
5. **规则冲突解决**：当多条规则冲突时决定执行顺序

## 二、核心组件

```
RuleModule
├── RuleRepository          # 规则仓储
├── RuleEngine              # 规则引擎
├── ConstraintEvaluator     # 约束评估器
├── TriggerManager          # 触发器管理器
└── ConsequenceExecutor     # 后果执行器
```

## 三、RuleEngine

### 3.1 核心逻辑

```python
class RuleEngine:
    """规则引擎"""
    
    def __init__(
        self,
        event_bus: EventBus,
        repository: RuleRepository
    ):
        self.event_bus = event_bus
        self.repository = repository
        self.active_rules: Dict[str, Rule] = {}  # 会话ID -> 规则列表
        self._setup_subscriptions()
    
    def _setup_subscriptions(self):
        """订阅相关事件"""
        self.event_bus.subscribe(EventType.MESSAGE_RECEIVED, self._on_message_received)
        self.event_bus.subscribe(EventType.SESSION_STARTED, self._on_session_started)
        self.event_bus.subscribe(EventType.SESSION_ENDED, self._on_session_ended)
```

```python
    def load_rules(self, session_id: str, templates: List[RuleTemplate]) -> List[Rule]:
        """加载规则到会话"""
        rules = []
        for template in templates:
            rule = Rule.from_template(template)
            rules.append(rule)
        
        self.active_rules[session_id] = rules
        
        # 发布规则加载事件
        for rule in rules:
            self.event_bus.publish(Event(
                type=EventType.RULE_LOADED,
                source="rule_engine",
                session_id=session_id,
                data={"rule_id": rule.id, "rule_name": rule.name}
            ))
        
        return rules
```

### 3.3 规则检查

```python
    async def check_rules(
        self,
        session_id: str,
        context: Dict[str, Any]
    ) -> RuleCheckResult:
        """
        检查所有规则
        
        返回：
        - satisfied_rules: 满足的规则列表
        - violated_rules: 违反的规则列表
        - triggered_rules: 触发的规则列表
        """
        if session_id not in self.active_rules:
            return RuleCheckResult()
        
        rules = self.active_rules[session_id]
        result = RuleCheckResult()
        
        for rule in rules:
            if not rule.enabled:
                continue
            
            # 检查约束
            constraint_result = self._evaluate_constraints(rule, context)
            if constraint_result.satisfied:
                result.satisfied_rules.append(rule)
            else:
                result.violated_rules.append(rule)
                self.event_bus.publish(Event(
                    type=EventType.RULE_VIOLATED,
                    source="rule_engine",
                    session_id=session_id,
                    data={
                        "rule_id": rule.id,
                        "violation": constraint_result.violation
                    }
                ))
            
            # 检查触发器
            trigger_result = self._check_triggers(rule, context)
            if trigger_result.triggered:
                result.triggered_rules.append(rule)
                self.event_bus.publish(Event(
                    type=EventType.RULE_TRIGGERED,
                    source="rule_engine",
                    session_id=session_id,
                    data={"rule_id": rule.id}
                ))
                # 执行后果
                await self._execute_consequences(rule, context)
        
        self.event_bus.publish(Event(
            type=EventType.RULE_CHECKED,
            source="rule_engine",
            session_id=session_id,
            data={"result": result.to_dict()}
        ))
        
        return result
```

## 四、ConstraintEvaluator

### 4.1 约束评估器

```python
class ConstraintEvaluator:
    """约束条件评估器"""
    
    def __init__(self):
        self.expression_parser = ExpressionParser()
    
    def evaluate(
        self,
        constraint: Constraint,
        context: Dict[str, Any]
    ) -> ConstraintResult:
        """评估约束条件"""
        try:
            # 解析表达式
            expression = self.expression_parser.parse(constraint.condition)
            
            # 构建评估上下文
            eval_context = self._build_context(constraint, context)
            
            # 执行评估
            satisfied = expression.evaluate(eval_context)
            
            return ConstraintResult(
                satisfied=satisfied,
                violation=None if satisfied else {
                    "constraint_id": constraint.id,
                    "message": constraint.error_message,
                    "details": f"条件 '{constraint.condition}' 不满足"
                }
            )
        except ExpressionError as e:
            return ConstraintResult(
                satisfied=False,
                violation={
                    "constraint_id": constraint.id,
                    "message": "约束评估错误",
                    "details": str(e)
                }
            )
    
    def _build_context(
        self,
        constraint: Constraint,
        context: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        构建评估上下文
        
        补充说明：完整的上下文应包含以下信息
        """
        eval_context = dict(context)
        
        # ==================== 消息上下文 ====================
        message = context.get("message", {})
        eval_context["message"] = message
        eval_context["message_content"] = message.get("content", "")
        eval_context["message_length"] = len(message.get("content", ""))
        eval_context["message_type"] = message.get("type", "text")
        eval_context["len"] = eval_context["message_length"]  # 向后兼容别名
        
        # ==================== 会话上下文 ====================
        session = context.get("session", {})
        eval_context["session"] = session
        eval_context["session_id"] = session.get("id", "")
        eval_context["session_state"] = session.get("state", "")
        eval_context["turn_count"] = context.get("turn_count", 0)
        eval_context["max_turns"] = session.get("config", {}).get("max_turns", float('inf'))
        
        # ==================== 参与者上下文 ====================
        participant = context.get("participant", {})
        eval_context["participant"] = participant
        eval_context["participant_id"] = participant.get("id", "")
        eval_context["participant_name"] = participant.get("name", "")
        eval_context["participant_type"] = participant.get("type", "ai")
        eval_context["is_ai"] = participant.get("is_ai", True)
        
        # ==================== 历史上下文 ====================
        history = context.get("history", [])
        eval_context["history"] = history
        eval_context["history_count"] = len(history)
        eval_context["recent_messages"] = history[-3:] if history else []
        
        # ==================== 规则上下文 ====================
        eval_context["rule_id"] = context.get("rule_id", "")
        eval_context["rule_name"] = context.get("rule_name", "")
        
        # ==================== 时间上下文 ====================
        from datetime import datetime
        eval_context["timestamp"] = datetime.now()
        
        return eval_context
```

### 4.2 支持的操作符

```python
class ExpressionParser:
    """表达式解析器"""
    
    # 允许的操作符
    OPERATORS = {
        "==": lambda a, b: a == b,
        "!=": lambda a, b: a != b,
        ">": lambda a, b: a > b,
        "<": lambda a, b: a < b,
        ">=": lambda a, b: a >= b,
        "<=": lambda a, b: a <= b,
        "in": lambda a, b: a in b,
        "not in": lambda a, b: a not in b,
        "and": lambda a, b: a and b,
        "or": lambda a, b: a or b,
    }
    
    # 允许的函数
    ALLOWED_FUNCTIONS = {"len", "str", "int", "float", "bool", "abs", "min", "max", "round"}
    
    # 危险模式（增强安全检查）
    DANGEROUS_PATTERNS = [
        r"__",                    # 双下划线（特殊方法）
        r"\beval\b",             # eval 函数
        r"\bexec\b",             # exec 函数
        r"\bimport\b",          # import 语句
        r"\bopen\b",             # 文件操作
        r"\bcompile\b",         # 编译函数
        r"os\.",                 # os 模块
        r"sys\.",                # sys 模块
        r"subprocess",           # 子进程
        r"requests",              # 网络请求
        r"\bglobals\b",         # 全局变量
        r"\blocals\b",           # 局部变量
        r"\bvars\b",            # 变量字典
        r"\bdir\b",             # 目录函数
        r"\[.*\].*=",            # 列表/字典修改
    ]
    
    def parse(self, expression: str) -> Expression:
        """解析表达式"""
        # 安全检查（增强版）
        if not self._is_safe(expression):
            raise ExpressionError(f"不安全的表达式: {expression}")
        
        # 解析为表达式树
        return self._parse_expression(expression)
    
    def _is_safe(self, expression: str) -> bool:
        """增强安全检查"""
        import re
        
        # 检查危险模式
        for pattern in self.DANGEROUS_PATTERNS:
            if re.search(pattern, expression, re.IGNORECASE):
                return False
        
        # 检查函数调用安全性
        func_pattern = r'\b([a-zA-Z_][a-zA-Z0-9_]*)\s*\('
        matches = re.findall(func_pattern, expression)
        for func_name in matches:
            if func_name.lower() not in self.ALLOWED_FUNCTIONS:
                return False
        
        return True
```

## 五、TriggerManager

### 5.1 触发器管理器

```python
class TriggerManager:
    """触发器管理器"""
    
    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self.active_triggers: Dict[str, List[Trigger]] = {}
    
    def check_triggers(
        self,
        session_id: str,
        event: Event,
        context: Dict[str, Any]
    ) -> List[TriggerResult]:
        """检查触发器"""
        results = []
        
        if session_id not in self.active_triggers:
            return results
        
        for trigger in self.active_triggers[session_id]:
            if self._matches_event(trigger, event):
                # 评估触发条件
                condition_result = self._evaluate_condition(trigger, context)
                
                results.append(TriggerResult(
                    trigger=trigger,
                    triggered=condition_result,
                    event=event
                ))
        
        return results
    
    def _matches_event(self, trigger: Trigger, event: Event) -> bool:
        """检查事件是否匹配"""
        return trigger.event_type == event.type
    
    def _evaluate_condition(self, trigger: Trigger, context: Dict[str, Any]) -> bool:
        """评估触发条件"""
        if trigger.condition == "true":
            return True
        
        # 解析并评估条件
        try:
            return self._eval_condition(trigger.condition, context)
        except Exception:
            return False
```

## 六、ConsequenceExecutor

### 6.1 后果执行器

```python
class ConsequenceExecutor:
    """后果执行器"""
    
    def __init__(self, event_bus: EventBus, ai_engine: AIEngine):
        self.event_bus = event_bus
        self.ai_engine = ai_engine
    
    async def execute(
        self,
        consequence: Consequence,
        context: Dict[str, Any]
    ) -> ConsequenceResult:
        """执行后果"""
        result = ConsequenceResult(
            success=False,
            action=consequence.action,
            details=None
        )
        
        try:
            if consequence.action == "send_message":
                result = await self._execute_send_message(consequence, context)
            elif consequence.action == "modify_behavior":
                result = await self._execute_modify_behavior(consequence, context)
            elif consequence.action == "apply_penalty":
                result = self._execute_apply_penalty(consequence, context)
            elif consequence.action == "end_session":
                result = await self._execute_end_session(consequence, context)
            
            result.success = True
        except Exception as e:
            result.details = str(e)
        
        return result
    
    async def _execute_send_message(
        self,
        consequence: Consequence,
        context: Dict[str, Any]
    ) -> ConsequenceResult:
        """执行发送消息"""
        await self.ai_engine.send_system_message(
            session_id=context["session_id"],
            content=consequence.params["content"]
        )
        return ConsequenceResult(success=True)
    
    async def _execute_modify_behavior(
        self,
        consequence: Consequence,
        context: Dict[str, Any]
    ) -> ConsequenceResult:
        """执行行为修改"""
        agent_id = consequence.params.get("agent_id")
        behavior = consequence.params.get("behavior")
        value = consequence.params.get("value")
        
        await self.ai_engine.modify_behavior(agent_id, behavior, value)
        return ConsequenceResult(success=True)
    
    def _execute_apply_penalty(
        self,
        consequence: Consequence,
        context: Dict[str, Any]
    ) -> ConsequenceResult:
        """执行惩罚"""
        participant_id = consequence.params.get("participant_id")
        penalty = consequence.params.get("penalty")
        
        # 增加惩罚计数
        context["penalties"] = context.get("penalties", {})
        context["penalties"][participant_id] = \
            context["penalties"].get(participant_id, 0) + penalty
        
        return ConsequenceResult(success=True)
    
    async def _execute_end_session(
        self,
        consequence: Consequence,
        context: Dict[str, Any]
    ) -> ConsequenceResult:
        """执行结束会话"""
        reason = consequence.params.get("reason", "规则违反")
        await self.event_bus.publish(Event(
            type=EventType.SESSION_ENDED,
            source="rule_engine",
            session_id=context["session_id"],
            data={"reason": reason}
        ))
        return ConsequenceResult(success=True)
```

## 七、RuleRepository

### 7.1 规则仓储

```python
class RuleRepository:
    """规则仓储"""
    
    def __init__(self, db: Database):
        self.db = db
        self._cache: Dict[str, RuleTemplate] = {}
    
    async def save(self, template: RuleTemplate) -> None:
        """保存规则模板"""
        await self.db.rules.update_one(
            {"_id": template.id},
            {"$set": template.to_dict()},
            upsert=True
        )
        self._cache[template.id] = template
    
    async def find_by_id(self, rule_id: str) -> Optional[RuleTemplate]:
        """根据ID查找"""
        if rule_id in self._cache:
            return self._cache[rule_id]
        
        doc = await self.db.rules.find_one({"_id": rule_id})
        if doc:
            template = RuleTemplate.from_dict(doc)
            self._cache[rule_id] = template
            return template
        return None
    
    async def find_all(self) -> List[RuleTemplate]:
        """查找所有"""
        docs = await self.db.rules.find().to_list(1000)
        return [RuleTemplate.from_dict(doc) for doc in docs]
    
    async def find_by_category(self, category: str) -> List[RuleTemplate]:
        """根据类别查找"""
        docs = await self.db.rules.find({"category": category}).to_list(1000)
        return [RuleTemplate.from_dict(doc) for doc in docs]
    
    async def delete(self, rule_id: str) -> bool:
        """删除规则"""
        result = await self.db.rules.delete_one({"_id": rule_id})
        if rule_id in self._cache:
            del self._cache[rule_id]
        return result.deleted_count > 0
```

### 9.1 事件处理

```python
class RuleEventHandler:
    """规则事件处理器"""
    
    def __init__(self, engine: RuleEngine):
        self.engine = engine
    
    async def handle_message_received(self, event: Event):
        """处理消息接收事件"""
        context = {
            "session_id": event.session_id,
            "message": event.data.get("message"),
            "participant": event.data.get("participant")
        }
        
        await self.engine.check_rules(event.session_id, context)
    
    async def handle_session_started(self, event: Event):
        """处理会话开始事件"""
        templates = event.data.get("rule_templates", [])
        self.engine.load_rules(event.session_id, templates)
    
    async def handle_session_ended(self, event: Event):
        """处理会话结束事件"""
        self.engine.unload_rules(event.session_id)
```
