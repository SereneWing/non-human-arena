# 心跳引擎设计

## 一、设计目标

心跳引擎是系统的主动行为触发模块，负责：
1. **定时触发**：按固定间隔触发心跳事件
2. **AI主动行为**：让空闲的AI角色主动思考或发言
3. **会话保活**：维护会话的活跃状态
4. **资源管理**：智能管理定时器资源

## 二、设计原则

1. **异步优先**：使用异步定时器，不阻塞主线程
2. **可配置间隔**：心跳间隔可动态调整
3. **按会话隔离**：每个会话独立的计时器
4. **低资源消耗**：心跳处理轻量化

## 四、核心接口

from typing import Dict, Optional, Callable, Awaitable
from dataclasses import dataclass
from enum import Enum

class HeartbeatPolicy(Enum):
    """心跳策略"""
    FIXED_RATE = "fixed_rate"        # 固定速率
    ADAPTIVE = "adaptive"            # 自适应（根据活跃度调整）

@dataclass
class HeartbeatConfig:
    """心跳配置"""
    interval_seconds: float = 5.0     # 心跳间隔（秒）
    policy: HeartbeatPolicy = HeartbeatPolicy.FIXED_RATE
    min_interval: float = 2.0        # 最小间隔（自适应模式）
    max_interval: float = 30.0       # 最大间隔（自适应模式）
    enabled: bool = True             # 是否启用

class IHeartbeatEngine(Protocol):
    """心跳引擎接口"""
    
    async def start(self, session_id: str) -> None:
        """启动会话的心跳"""
        ...
    
    async def stop(self, session_id: str) -> None:
        """停止会话的心跳"""
        ...
    
    async def pause(self, session_id: str) -> None:
        """暂停心跳"""
        ...
    
    async def resume(self, session_id: str) -> None:
        """恢复心跳"""
        ...
    
    def get_config(self, session_id: str) -> Optional[HeartbeatConfig]:
        """获取心跳配置"""
        ...
    
    async def set_config(self, session_id: str, config: HeartbeatConfig) -> None:
        """设置心跳配置"""
        ...

## 五、核心功能

### 5.1 心跳事件发布

```python
class HeartbeatEngine:
    """心跳引擎实现"""
    
    def __init__(
        self,
        event_bus: IEventBus,
        config: HeartbeatConfig = None
    ):
        self.event_bus = event_bus
        self.default_config = config or HeartbeatConfig()
        self._timers: Dict[str, asyncio.TimerHandle] = {}
        self._paused: Dict[str, bool] = {}
        self._session_configs: Dict[str, HeartbeatConfig] = {}
    
    async def start(self, session_id: str) -> None:
        """启动心跳"""
        if session_id in self._timers:
            return
        
        config = self._session_configs.get(session_id, self.default_config)
        if not config.enabled:
            return
        
        self._paused[session_id] = False
        await self._schedule_next(session_id, config)
    
    async def stop(self, session_id: str) -> None:
        """停止心跳"""
        if session_id in self._timers:
            self._timers[session_id].cancel()
            del self._timers[session_id]
        
        self._paused.pop(session_id, None)
        self._session_configs.pop(session_id, None)
    
    async def _tick(self, session_id: str) -> None:
        """心跳触发"""
        config = self._session_configs.get(session_id, self.default_config)
        
        # 发布心跳事件
        await self.event_bus.publish(Event(
            type=EventType.AI_HEARTBEAT,
            source="heartbeat_engine",
            session_id=session_id,
            data={
                "timestamp": datetime.now().isoformat(),
                "interval": config.interval_seconds
            }
        ))
        
        # 调度下一次
        await self._schedule_next(session_id, config)
    
    async def _schedule_next(self, session_id: str, config: HeartbeatConfig) -> None:
        """调度下一次心跳"""
        if session_id not in self._timers:
            return
        
        loop = asyncio.get_event_loop()
        self._timers[session_id] = loop.call_later(
            config.interval_seconds,
            lambda: asyncio.create_task(self._safe_tick(session_id))
        )
```

### 5.2 自适应心跳

```python
class AdaptiveHeartbeatEngine(HeartbeatEngine):
    """自适应心跳引擎"""
    
    def __init__(self, event_bus: IEventBus):
        super().__init__(event_bus)
        self._activity_scores: Dict[str, float] = {}  # 活跃度分数
    
    async def _calculate_interval(self, session_id: str) -> float:
        """根据活跃度计算间隔"""
        config = self._session_configs.get(session_id, self.default_config)
        activity = self._activity_scores.get(session_id, 0.5)
        
        # 活跃度高 -> 减少间隔
        # 活跃度低 -> 增加间隔
        factor = 1.0 - activity * 0.5  # 0.5 ~ 1.0
        interval = config.min_interval + (config.max_interval - config.min_interval) * factor
        
        return max(config.min_interval, min(interval, config.max_interval))
    
    async def record_activity(self, session_id: str) -> None:
        """记录活动（被调用时增加活跃度）"""
        current = self._activity_scores.get(session_id, 0.0)
        self._activity_scores[session_id] = min(1.0, current + 0.2)
    
    async def _tick(self, session_id: str) -> None:
        """心跳触发（自适应版本）"""
        # 降低活跃度
        current = self._activity_scores.get(session_id, 0.0)
        self._activity_scores[session_id] = max(0.0, current - 0.1)
        
        # 使用自适应间隔
        config = self._session_configs.get(session_id, self.default_config)
        config.interval_seconds = await self._calculate_interval(session_id)
        
        await super()._tick(session_id)
```

## 六、与AI引擎的集成

```python
class HeartbeatAIManager:
    """心跳AI管理器"""
    
    def __init__(
        self,
        event_bus: IEventBus,
        heartbeat_engine: IHeartbeatEngine,
        decision_engine: IAIDecisionEngine
    ):
        self.event_bus = event_bus
        self.heartbeat_engine = heartbeat_engine
        self.decision_engine = decision_engine
        self._idle_agents: Dict[str, List[str]] = {}  # session_id -> [agent_ids]
        
        # 订阅心跳事件
        event_bus.subscribe(EventType.AI_HEARTBEAT, self.on_heartbeat)
        event_bus.subscribe(EventType.MESSAGE_SENT, self.on_message_sent)
    
    async def register_agent(self, session_id: str, agent_id: str) -> None:
        """注册AI代理"""
        if session_id not in self._idle_agents:
            self._idle_agents[session_id] = []
        self._idle_agents[session_id].append(agent_id)
    
    async def on_heartbeat(self, event: Event) -> None:
        """处理心跳事件"""
        session_id = event.session_id
        if session_id not in self._idle_agents:
            return
        
        # 选择一个空闲的代理
        agents = self._idle_agents[session_id]
        if not agents:
            return
        
        agent_id = agents.pop(0)
        
        # 让AI决策是否发言
        decision = await self.decision_engine.should_speak(
            agent_id=agent_id,
            context=event.data
        )
        
        if decision.should_speak:
            # AI发言
            await self.decision_engine.generate_response(
                agent_id=agent_id,
                context=decision.reasoning
            )
        else:
            # 放回空闲队列
            agents.append(agent_id)
    
    async def on_message_sent(self, event: Event) -> None:
        """处理消息发送事件"""
        session_id = event.session_id
        
        # 有新消息时重置AI的思考
        sender_id = event.data.get("sender_id")
        if sender_id in self._idle_agents.get(session_id, []):
            self._idle_agents[session_id].remove(sender_id)
```

## 七、使用示例

```python
# 启动心跳
heartbeat_engine = HeartbeatEngine(event_bus)
await heartbeat_engine.start("session_001")

# 配置心跳参数
await heartbeat_engine.set_config("session_001", HeartbeatConfig(
    interval_seconds=10.0,
    policy=HeartbeatPolicy.ADAPTIVE
))

# 暂停/恢复
await heartbeat_engine.pause("session_001")
await heartbeat_engine.resume("session_001")

# 停止心跳
await heartbeat_engine.stop("session_001")
```

## 八、配置建议

| 场景 | 间隔 | 策略 | 说明 |
|------|------|------|------|
| 实时辩论 | 3-5秒 | FIXED_RATE | 需要快速响应 |
| 角色扮演 | 10-15秒 | ADAPTIVE | 根据对话活跃度调整 |
| 狼人杀 | 5-10秒 | FIXED_RATE | 需要定时器精确 |
| 安静观察 | 20-30秒 | ADAPTIVE | 减少不必要的AI思考 |
