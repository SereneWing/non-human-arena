### 5.1 WebSocket 重连策略

```typescript
// 5.1 WebSocket 重连策略

interface ReconnectionPolicy {
  maxAttempts: number      // 最大重连次数
  initialDelay: number      // 初始延迟（毫秒）
  maxDelay: number          // 最大延迟（毫秒）
  backoffMultiplier: number // 退避乘数
}

const DEFAULT_RECONNECTION_POLICY: ReconnectionPolicy = {
  maxAttempts: 5,
  initialDelay: 1000,       // 1秒
  maxDelay: 30000,          // 30秒
  backoffMultiplier: 2.0
}

class WebSocketManager {
  private ws: WebSocket | null = null
  private reconnectAttempts = 0
  private reconnectTimer: number | null = null
  private policy: ReconnectionPolicy
  
  // 回调
  private onOpen: () => void
  private onMessage: (data: any) => void
  private onClose: () => void
  private onError: (error: Event) => void
  
  constructor(
    url: string,
    callbacks: {
      onOpen: () => void
      onMessage: (data: any) => void
      onClose: () => void
      onError: (error: Event) => void
    },
    policy: ReconnectionPolicy = DEFAULT_RECONNECTION_POLICY
  ) {
    this.onOpen = callbacks.onOpen
    this.onMessage = callbacks.onMessage
    this.onClose = callbacks.onClose
    this.onError = callbacks.onError
    this.policy = policy
    
    this.connect(url)
  }
  
  connect(url: string) {
    this.ws = new WebSocket(url)
    
    this.ws.onopen = () => {
      this.reconnectAttempts = 0
      this.onOpen()
    }
    
    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data)
      this.onMessage(data)
    }
    
    this.ws.onclose = (event) => {
      this.onClose()
      
      // 检查是否是主动关闭
      if (!event.wasClean) {
        this.scheduleReconnect(url)
      }
    }
    
    this.ws.onerror = (error) => {
      this.onError(error)
    }
  }
  
  private scheduleReconnect(url: string) {
    if (this.reconnectAttempts >= this.policy.maxAttempts) {
      console.error('Max reconnection attempts reached')
      return
    }
    
    // 计算延迟（指数退避）
    const delay = Math.min(
      this.policy.initialDelay * Math.pow(this.policy.backoffMultiplier, this.reconnectAttempts),
      this.policy.maxDelay
    )
    
    console.log(`Scheduling reconnect in ${delay}ms (attempt ${this.reconnectAttempts + 1})`)
    
    this.reconnectTimer = window.setTimeout(() => {
      this.reconnectAttempts++
      this.connect(url)
    }, delay)
  }
  
  send(data: any) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(data))
      return true
    }
    return false
  }
  
  disconnect() {
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer)
      this.reconnectTimer = null
    }
    
    if (this.ws) {
      this.ws.close()
      this.ws = null
    }
  }
  
  isConnected() {
    return this.ws?.readyState === WebSocket.OPEN
  }
}
```

### 5.2 会话状态管理

```typescript
// stores/session.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { WebSocketManager, DEFAULT_RECONNECTION_POLICY } from '@/utils/websocket'

export const useSessionStore = defineStore('session', () => {
  const session = ref<Session | null>(null)
  const messages = ref<Message[]>([])
  const participants = ref<Participant[]>([])
  const isConnected = ref(false)
  const isReconnecting = ref(false)
  const connectionError = ref<string | null>(null)
  
  let wsManager: WebSocketManager | null = null
  
  // 计算属性
  const canSend = computed(() => 
    isConnected.value && session.value?.state === 'running'
  )
  
  const connectionStatus = computed(() => {
    if (isReconnecting.value) return 'reconnecting'
    if (isConnected.value) return 'connected'
    if (connectionError.value) return 'error'
    return 'disconnected'
  })
  
  const connectWebSocket = (sessionId: string) => {
    const wsUrl = `ws://localhost:8000/ws/${sessionId}`
    
    wsManager = new WebSocketManager(
      wsUrl,
      {
        onOpen: () => {
          isConnected.value = true
          isReconnecting.value = false
          connectionError.value = null
          console.log('WebSocket connected')
        },
        
        onMessage: (data) => {
          handleWebSocketMessage(data)
        },
        
        onClose: () => {
          isConnected.value = false
          console.log('WebSocket closed')
        },
        
        onError: (error) => {
          isReconnecting.value = true
          connectionError.value = '连接失败，正在重连...'
          console.error('WebSocket error:', error)
        }
      },
      {
        ...DEFAULT_RECONNECTION_POLICY,
        maxAttempts: 10  // 增加重连次数
      }
    )
  }
  
  const handleWebSocketMessage = (data: any) => {
    switch (data.type) {
      case 'message':
        messages.value.push(data.payload)
        break
        
      case 'participant_joined':
        participants.value.push(data.payload)
        break
        
      case 'participant_left':
        participants.value = participants.value.filter(
          p => p.id !== data.payload.id
        )
        break
        
      case 'session_state_changed':
        session.value.state = data.payload.state
        break
        
      case 'heartbeat':
        // 心跳响应
        break
        
      case 'error':
        console.error('Server error:', data.payload)
        connectionError.value = data.payload.message
        break
    }
  }
  
  const sendMessage = async (content: string) => {
    if (!canSend.value) {
      console.warn('Cannot send message: not connected or session not running')
      return false
    }
    
    const sent = wsManager?.send({
      type: 'message',
      content,
      sender_id: 'user',
      timestamp: Date.now()
    })
    
    if (!sent) {
      connectionError.value = '发送失败，连接已断开'
      return false
    }
    
    return true
  }
  
  const disconnect = () => {
    wsManager?.disconnect()
    wsManager = null
    isConnected.value = false
    isReconnecting.value = false
  }
  
  const clearSession = () => {
    disconnect()
    session.value = null
    messages.value = []
    participants.value = []
    connectionError.value = null
  }
  
  return {
    // 状态
    session,
    messages,
    participants,
    isConnected,
    isReconnecting,
    connectionError,
    
    // 计算属性
    canSend,
    connectionStatus,
    
    // 方法
    connectWebSocket,
    sendMessage,
    disconnect,
    clearSession
  }
})
```

```typescript
// api/client.ts
const API_BASE_URL = 'http://localhost:8000/api/v1'

class ApiClient {
  private async request<T>(
    endpoint: string,
    options: RequestInit = {}
  ): Promise<T> {
    const response = await fetch(`${API_BASE_URL}${endpoint}`, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...options.headers
      }
    })
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`)
    }
    
    return response.json()
  }
  
  // 角色接口
  async listRoles(params?: { skip?: number; limit?: number }) {
    const query = new URLSearchParams(params as any).toString()
    return this.request<any>(`/roles?${query}`)
  }
  
  async createRole(data: any) {
    return this.request<any>('/roles', {
      method: 'POST',
      body: JSON.stringify(data)
    })
  }
  
  async getRole(id: string) {
    return this.request<any>(`/roles/${id}`)
  }
  
  async updateRole(id: string, data: any) {
    return this.request<any>(`/roles/${id}`, {
      method: 'PUT',
      body: JSON.stringify(data)
    })
  }
  
  async deleteRole(id: string) {
    return this.request<void>(`/roles/${id}`, {
      method: 'DELETE'
    })
  }
  
  // 会话接口
  async createSession(data: any) {
    return this.request<any>('/sessions', {
      method: 'POST',
      body: JSON.stringify(data)
    })
  }
  
  async startSession(id: string) {
    return this.request<any>(`/sessions/${id}/start`, {
      method: 'POST'
    })
  }
  
  async endSession(id: string) {
    return this.request<any>(`/sessions/${id}/end`, {
      method: 'POST'
    })
  }
}

export const apiClient = new ApiClient()
```
