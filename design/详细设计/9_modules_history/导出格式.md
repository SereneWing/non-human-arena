## 六、导出服务实现

```python
class ExportService:
    """导出服务"""
    
    SUPPORTED_FORMATS = ["json", "markdown", "html", "text"]
    
    async def export_session(
        self,
        session_id: str,
        format: str = "json",
        include_mental: bool = True
    ) -> str:
        """导出会话"""
        session = await self.session_service.get_session(session_id)
        messages = await self.history_service.get_history(session_id)
        
        data = self._build_export_data(session, messages, include_mental)
        
        if format == "json":
            return json.dumps(data, ensure_ascii=False, indent=2)
        elif format == "markdown":
            return self._to_markdown(data)
        elif format == "html":
            return self._to_html(data)
        elif format == "text":
            return self._to_text(data)
```

## 七、导出 API

```python
@router.get("/sessions/{session_id}/export")
async def export_session(
    session_id: str,
    format: str = Query("json", regex="^(json|markdown|html|text)$")
):
    """导出会话历史"""
    content = await export_service.export_session(session_id, format)
    
    content_types = {
        "json": "application/json",
        "markdown": "text/markdown",
        "html": "text/html",
        "text": "text/plain"
    }
    
    return Response(
        content=content,
        media_type=content_types[format],
        headers={"Content-Disposition": f"attachment; filename=session_{session_id}.{format}"}
    )
```
