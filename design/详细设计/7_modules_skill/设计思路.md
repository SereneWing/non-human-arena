# 技能模块设计

## 一、设计目标

技能模块是系统的扩展模块之一，负责：
1. **技能注册**：管理AI可用的技能
2. **技能执行**：执行技能的逻辑
3. **技能触发**：根据上下文触发相应技能

## 二、设计原则

1. **插件化**：技能可热插拔
2. **声明式**：技能定义声明式
3. **可组合**：技能可组合使用

## 五、技能执行器实现

```python
class SkillExecutor:
    """技能执行器"""
    
    def __init__(
        self,
        registry: ISkillRegistry,
        event_bus: IEventBus
    ):
        self.registry = registry
        self.event_bus = event_bus
    
    async def execute(
        self,
        skill_id: str,
        context: Dict[str, Any]
    ) -> SkillResult:
        """执行技能"""
        skill = await self.registry.get(skill_id)
        if not skill:
            raise SkillNotFoundError(skill_id)
        
        # 检查前置条件
        if not await self.can_execute(skill_id, context):
            return SkillResult(
                success=False,
                skill_id=skill_id,
                error="Preconditions not met"
            )
        
        try:
            # 发布技能触发事件
            await self.event_bus.publish(create_event(
                event_type=EventType.SKILL_TRIGGERED,
                source="skill_executor",
                session_id=context.get("session_id"),
                data={"skill_id": skill_id, "context": context}
            ))
            
            # 执行技能逻辑
            result = await skill.execute(context)
            
            # 发布技能结果事件
            await self.event_bus.publish(create_event(
                event_type=EventType.SKILL_RESULT,
                source="skill_executor",
                session_id=context.get("session_id"),
                data={"skill_id": skill_id, "result": result}
            ))
            
            return SkillResult(
                success=True,
                skill_id=skill_id,
                result=result
            )
        
        except Exception as e:
            logger.error(f"Skill execution failed: {e}")
            return SkillResult(
                success=False,
                skill_id=skill_id,
                error=str(e)
            )
    
    async def can_execute(
        self,
        skill_id: str,
        context: Dict[str, Any]
    ) -> bool:
        """检查是否可以执行"""
        skill = await self.registry.get(skill_id)
        if not skill:
            return False
        
        # 检查前置条件
        for condition in skill.preconditions:
            if not await self._check_condition(condition, context):
                return False
        
        return True
    
    async def _check_condition(
        self,
        condition: SkillCondition,
        context: Dict[str, Any]
    ) -> bool:
        """检查条件"""
        value = context.get(condition.field)
        
        if condition.operator == "exists":
            return value is not None
        elif condition.operator == "equals":
            return value == condition.value
        elif condition.operator == "contains":
            return condition.value in (value or [])
        
        return True
```
