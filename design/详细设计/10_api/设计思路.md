```python
# v1/roles.py
@router.get("/roles")
async def list_roles(
    skip: int = 0,
    limit: int = 100,
    category: str = None,
    role_service: IRoleService = Depends(get_role_service)
):
    """列出角色"""
    return await role_service.list_roles(skip, limit)

@router.post("/roles")
async def create_role(
    role_data: RoleCreate,
    role_service: IRoleService = Depends(get_role_service)
):
    """创建角色"""
    return await role_service.create_role(role_data.dict())

@router.get("/roles/{role_id}")
async def get_role(
    role_id: str,
    role_service: IRoleService = Depends(get_role_service)
):
    """获取角色"""
    role = await role_service.get_role(role_id)
    if not role:
        raise HTTPException(status_code=404, detail="Role not found")
    return role

@router.put("/roles/{role_id}")
async def update_role(
    role_id: str,
    role_data: RoleUpdate,
    role_service: IRoleService = Depends(get_role_service)
):
    """更新角色"""
    return await role_service.update_role(role_id, role_data.dict())

@router.delete("/roles/{role_id}")
async def delete_role(
    role_id: str,
    role_service: IRoleService = Depends(get_role_service)
):
    """删除角色"""
    await role_service.delete_role(role_id)
    return {"success": True}
```

## 四、WebSocket处理

```python
# websocket.py
class ConnectionManager:
    """连接管理器"""
    
    def __init__(self):
        self.active_connections: Dict[str, WebSocket] = {}
    
    async def connect(self, websocket: WebSocket, session_id: str):
        """连接"""
        await websocket.accept()
        self.active_connections[session_id] = websocket
    
    def disconnect(self, session_id: str):
        """断开连接"""
        if session_id in self.active_connections:
            del self.active_connections[session_id]
    
    async def send_message(self, session_id: str, message: dict):
        """发送消息"""
        if session_id in self.active_connections:
            await self.active_connections[session_id].send_json(message)


manager = ConnectionManager()

@router.websocket("/ws/{session_id}")
async def websocket_endpoint(
    websocket: WebSocket,
    session_id: str
):
    """WebSocket端点"""
    await manager.connect(websocket, session_id)
    
    try:
        while True:
            data = await websocket.receive_json()
            
            # 处理消息
            if data.get("type") == "message":
                message = await session_service.send_message(
                    session_id=session_id,
                    content=data.get("content"),
                    sender_id=data.get("sender_id")
                )
            elif data.get("type") == "command":
                # 处理命令
                pass
    
    except WebSocketDisconnect:
        manager.disconnect(session_id)
```
