# 角色模块设计

## 一、设计目标

角色模块是系统的基础模块之一，负责：
1. **角色管理**：角色的创建、读取、更新、删除
2. **角色模板**：预置角色模板和自定义模板
3. **角色继承**：支持基于模板创建角色
4. **角色实例化**：在会话中创建角色实例

## 二、设计原则

1. **接口抽象**：通过仓储接口抽象数据访问
2. **事件驱动**：角色变更通过事件总线通知
3. **模板支持**：支持角色模板的继承和组合
4. **验证完整**：角色定义需满足完整性约束

## 五、服务层实现

```python
class RoleService:
    """角色服务实现"""
    
    def __init__(
        self,
        repository: IRoleRepository,
        event_bus: IEventBus,
        template_service: IRoleTemplateService
    ):
        self.repository = repository
        self.event_bus = event_bus
        self.template_service = template_service
    
    async def create_role(self, role_data: Dict[str, Any]) -> Role:
        """创建角色"""
        # 验证数据
        errors = await self.validate_role(role_data)
        if errors:
            raise RoleValidationError(errors)
        
        # 创建角色
        role = Role.from_dict(role_data)
        role = await self.repository.create(role)
        
        # 发布事件
        await self.event_bus.publish(create_role_event(
            event_type=EventType.ROLE_CREATED,
            role_id=role.id,
            role_data=role.to_dict()
        ))
        
        return role
    
    async def update_role(
        self,
        role_id: str,
        role_data: Dict[str, Any]
    ) -> Role:
        """更新角色"""
        # 检查角色是否存在
        existing = await self.repository.get_by_id(role_id)
        if not existing:
            raise RoleNotFoundError(role_id)
        
        # 验证数据
        errors = await self.validate_role(role_data)
        if errors:
            raise RoleValidationError(errors)
        
        # 更新角色
        role = Role.from_dict({**existing.to_dict(), **role_data})
        role.updated_at = datetime.now()
        role = await self.repository.update(role)
        
        # 发布事件
        await self.event_bus.publish(create_role_event(
            event_type=EventType.ROLE_UPDATED,
            role_id=role.id,
            role_data=role.to_dict()
        ))
        
        return role
    
    async def delete_role(self, role_id: str) -> bool:
        """删除角色"""
        # 检查角色是否存在
        existing = await self.repository.get_by_id(role_id)
        if not existing:
            raise RoleNotFoundError(role_id)
        
        # 删除角色
        result = await self.repository.delete(role_id)
        
        if result:
            # 发布事件
            await self.event_bus.publish(create_role_event(
                event_type=EventType.ROLE_DELETED,
                role_id=role_id,
                role_data={"deleted_at": datetime.now().isoformat()}
            ))
        
        return result
    
    async def create_from_template(
        self,
        template_id: str,
        name: str,
        overrides: Optional[Dict[str, Any]] = None
    ) -> Role:
        """从模板创建角色"""
        # 获取模板
        template = await self.template_service.get_template(template_id)
        if not template:
            raise TemplateNotFoundError(template_id)
        
        # 应用覆盖
        role_data = template.to_dict()
        if overrides:
            role_data = {**role_data, **overrides}
        
        # 设置新角色的基本信息
        role_data["id"] = f"role_{uuid.uuid4().hex[:8]}"
        role_data["name"] = name
        role_data["created_at"] = datetime.now().isoformat()
        role_data["updated_at"] = datetime.now().isoformat()
        
        # 创建角色
        return await self.create_role(role_data)
    
    async def validate_role(self, role_data: Dict[str, Any]) -> List[str]:
        """验证角色数据"""
        errors = []
        
        # 验证必需字段
        if not role_data.get("name"):
            errors.append("角色名称不能为空")
        
        # 验证类别
        category = role_data.get("category")
        if category and category not in [c.value for c in RoleCategory]:
            errors.append(f"无效的角色类别: {category}")
        
        # 验证性格特征范围
        personality = role_data.get("base_personality", {})
        for key, value in personality.items():
            if not isinstance(value, (int, float)):
                errors.append(f"性格特征 {key} 必须是数字")
            elif not 0 <= value <= 1:
                errors.append(f"性格特征 {key} 必须在 0-1 范围内")
        
        # 验证LLM参数
        params = role_data.get("parameters", {})
        if "temperature" in params and not 0 <= params["temperature"] <= 2:
            errors.append("temperature 必须在 0-2 范围内")
        
        return errors
```

```python
class SQLAlchemyRoleRepository:
    """SQLAlchemy角色仓储实现"""
    
    def __init__(self, db_session: AsyncSession):
        self.db = db_session
    
    async def create(self, role: Role) -> Role:
        """创建角色"""
        db_role = RoleModel(
            id=role.id,
            name=role.name,
            category=role.category.value if isinstance(role.category, RoleCategory) else role.category,
            description=role.description,
            avatar=role.avatar,
            base_personality=json.dumps(role.base_personality),
            system_prompt=role.system_prompt,
            mental_config=json.dumps(role.mental_config.to_dict()) if role.mental_config else None,
            emotional_model=json.dumps(role.emotional_model.to_dict()) if role.emotional_model else None,
            speaking_style=json.dumps(role.speaking_style.to_dict()) if role.speaking_style else None,
            skills=json.dumps(role.skills),
            parameters=json.dumps(role.parameters),
            variables=json.dumps(role.variables),
            created_at=role.created_at,
            updated_at=role.updated_at
        )
        
        self.db.add(db_role)
        await self.db.commit()
        await self.db.refresh(db_role)
        
        return self._to_entity(db_role)
    
    async def get_by_id(self, role_id: str) -> Optional[Role]:
        """根据ID获取角色"""
        result = await self.db.get(RoleModel, role_id)
        return self._to_entity(result) if result else None
    
    async def list_all(self, skip: int = 0, limit: int = 100) -> List[Role]:
        """列出所有角色"""
        result = await self.db.execute(
            select(RoleModel)
            .offset(skip)
            .limit(limit)
            .order_by(RoleModel.created_at.desc())
        )
        return [self._to_entity(r) for r in result.scalars().all()]
    
    async def update(self, role: Role) -> Role:
        """更新角色"""
        result = await self.db.execute(
            select(RoleModel).where(RoleModel.id == role.id)
        )
        db_role = result.scalar_one()
        
        for key, value in role.to_dict().items():
            if hasattr(db_role, key) and key != "id":
                setattr(db_role, key, value)
        
        await self.db.commit()
        await self.db.refresh(db_role)
        
        return self._to_entity(db_role)
    
    async def delete(self, role_id: str) -> bool:
        """删除角色"""
        result = await self.db.execute(
            select(RoleModel).where(RoleModel.id == role_id)
        )
        db_role = result.scalar_one_or_none()
        
        if db_role:
            await self.db.delete(db_role)
            await self.db.commit()
            return True
        
        return False
    
    def _to_entity(self, db_role: RoleModel) -> Role:
        """转换为实体"""
        return Role(
            id=db_role.id,
            name=db_role.name,
            category=RoleCategory(db_role.category),
            description=db_role.description,
            avatar=db_role.avatar,
            base_personality=json.loads(db_role.base_personality),
            system_prompt=db_role.system_prompt,
            mental_config=MentalConfig.from_dict(json.loads(db_role.mental_config)) if db_role.mental_config else None,
            emotional_model=EmotionalModel.from_dict(json.loads(db_role.emotional_model)) if db_role.emotional_model else None,
            speaking_style=SpeakingStyle.from_dict(json.loads(db_role.speaking_style)) if db_role.speaking_style else None,
            skills=json.loads(db_role.skills),
            parameters=json.loads(db_role.parameters),
            variables=json.loads(db_role.variables),
            created_at=db_role.created_at,
            updated_at=db_role.updated_at
        )
```

## 七、事件订阅

```python
class RoleEventHandler:
    """角色事件处理器"""
    
    def __init__(self, service: IRoleService):
        self.service = service
    
    @on_event(EventType.SESSION_CREATED)
    async def on_session_created(self, event: Event) -> None:
        """会话创建时验证角色"""
        session_data = event.data
        role_ids = session_data.get("role_ids", [])
        
        for role_id in role_ids:
            role = await self.service.get_role(role_id)
            if not role:
                raise RoleNotFoundError(role_id)
    
    @on_event(EventType.ROLE_LOADED)
    async def on_role_loaded(self, event: Event) -> None:
        """角色加载到会话"""
        logger.info(f"Role {event.data.get('role_id')} loaded to session {event.session_id}")
```

```sql
CREATE TABLE roles (
    id VARCHAR(64) PRIMARY KEY,
    name VARCHAR(128) NOT NULL,
    category VARCHAR(32) NOT NULL DEFAULT 'custom',
    description TEXT,
    avatar VARCHAR(512),
    base_personality JSON NOT NULL DEFAULT '{}',
    system_prompt TEXT,
    mental_config JSON,
    emotional_model JSON,
    speaking_style JSON,
    skills JSON NOT NULL DEFAULT '[]',
    parameters JSON NOT NULL DEFAULT '{}',
    variables JSON NOT NULL DEFAULT '{}',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_name (name),
    INDEX idx_category (category),
    INDEX idx_created_at (created_at)
);
```

## 十、异常定义

```python
class RoleError(Exception):
    """角色基础异常"""
    pass

class RoleNotFoundError(RoleError):
    """角色不存在"""
    def __init__(self, role_id: str):
        self.role_id = role_id
        super().__init__(f"Role not found: {role_id}")

class RoleValidationError(RoleError):
    """角色验证错误"""
    def __init__(self, errors: List[str]):
        self.errors = errors
        super().__init__(f"Role validation failed: {', '.join(errors)}")

class TemplateNotFoundError(RoleError):
    """模板不存在"""
    def __init__(self, template_id: str):
        self.template_id = template_id
        super().__init__(f"Template not found: {template_id}")
```
