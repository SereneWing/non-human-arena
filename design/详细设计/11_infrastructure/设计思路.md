## 三、数据库连接

```python
# database.py
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker
from sqlalchemy.orm import declarative_base

Base = declarative_base()

class DatabaseManager:
    """数据库管理器"""
    
    def __init__(self, database_url: str):
        self.engine = create_async_engine(database_url, echo=False)
        self.session_factory = async_sessionmaker(
            self.engine,
            class_=AsyncSession,
            expire_on_commit=False
        )
    
    async def get_session(self) -> AsyncSession:
        """获取会话"""
        async with self.session_factory() as session:
            yield session
    
    async def create_tables(self):
        """创建表"""
        async with self.engine.begin() as conn:
            await conn.run_sync(Base.metadata.create_all)
    
    async def close(self):
        """关闭连接"""
        await self.engine.dispose()
```

## 四、Redis连接

```python
# redis.py
import redis.asyncio as redis
from typing import Optional

class RedisManager:
    """Redis管理器"""
    
    def __init__(self, redis_url: str):
        self.redis_url = redis_url
        self._client: Optional[redis.Redis] = None
    
    async def connect(self):
        """连接"""
        self._client = redis.from_url(self.redis_url)
    
    async def close(self):
        """关闭连接"""
        if self._client:
            await self._client.close()
    
    @property
    def client(self) -> redis.Redis:
        """获取客户端"""
        if not self._client:
            raise RuntimeError("Redis not connected")
        return self._client
    
    async def get(self, key: str) -> Optional[str]:
        """获取值"""
        return await self.client.get(key)
    
    async def set(
        self,
        key: str,
        value: str,
        expire: int = None
    ):
        """设置值"""
        await self.client.set(key, value, ex=expire)
    
    async def delete(self, key: str):
        """删除"""
        await self.client.delete(key)
    
    async def publish(self, channel: str, message: str):
        """发布消息"""
        await self.client.publish(channel, message)
    
    async def subscribe(self, channel: str):
        """订阅"""
        pubsub = self.client.pubsub()
        await pubsub.subscribe(channel)
        return pubsub
```

## 五、文件存储

```python
# storage.py
import os
from pathlib import Path
from typing import Optional
import aiofiles

class FileStorage:
    """文件存储"""
    
    def __init__(self, base_path: str):
        self.base_path = Path(base_path)
        self.base_path.mkdir(parents=True, exist_ok=True)
    
    async def save(
        self,
        filename: str,
        content: bytes,
        path: str = ""
    ) -> str:
        """保存文件"""
        file_path = self.base_path / path / filename
        file_path.parent.mkdir(parents=True, exist_ok=True)
        
        async with aiofiles.open(file_path, 'wb') as f:
            await f.write(content)
        
        return str(file_path)
    
    async def read(self, filepath: str) -> Optional[bytes]:
        """读取文件"""
        try:
            file_path = self.base_path / filepath
            async with aiofiles.open(file_path, 'rb') as f:
                return await f.read()
        except FileNotFoundError:
            return None
    
    async def delete(self, filepath: str) -> bool:
        """删除文件"""
        try:
            file_path = self.base_path / filepath
            file_path.unlink()
            return True
        except FileNotFoundError:
            return False
    
    def get_url(self, filepath: str) -> str:
        """获取文件URL"""
        return f"/files/{filepath}"
```
