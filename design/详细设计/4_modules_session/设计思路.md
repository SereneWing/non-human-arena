# 会话模块设计

## 一、设计目标

会话模块是系统的核心模块之一，负责：
1. **会话生命周期管理**：创建、启动、暂停、恢复、结束会话
2. **参与者管理**：管理会话中的角色和用户
3. **消息路由**：处理消息的发送和分发
4. **状态维护**：维护会话的运行状态

## 二、设计原则

1. **事件驱动**：会话状态变化通过事件总线通知
2. **状态机管理**：使用状态机管理会话状态转换
3. **消息队列**：使用消息队列处理高并发消息
4. **持久化支持**：会话数据支持持久化存储

### 5.3 状态转换实现

```python
class SessionStateManager:
    """会话状态管理器"""
    
    def __init__(
        self,
        session_id: str,
        event_bus: IEventBus,
        repository: ISessionRepository
    ):
        self.session_id = session_id
        self.event_bus = event_bus
        self.repository = repository
        self._state = SessionState.CREATED
        self._state_history: List[Tuple[SessionState, datetime]] = []
    
    @property
    def state(self) -> SessionState:
        """获取当前状态"""
        return self._state
    
    async def transition(self, target_state: SessionState) -> bool:
        """状态转换"""
        if not SessionStateMachine.can_transition(self._state, target_state):
            logger.warning(
                f"Invalid state transition: {self._state} -> {target_state}"
            )
            return False
        
        old_state = self._state
        self._state = target_state
        self._state_history.append((target_state, datetime.now()))
        
        # 更新数据库
        session = await self.repository.get_by_id(self.session_id)
        if session:
            session.state = target_state
            await self.repository.update(session)
        
        # 发布状态变更事件
        await self.event_bus.publish(create_session_event(
            event_type=EventType.SESSION_STATE_CHANGED,
            session_id=self.session_id,
            data={
                "old_state": old_state.value,
                "new_state": target_state.value
            }
        ))
        
        return True
    
    async def start(self) -> bool:
        """开始会话"""
        return await self.transition(SessionState.RUNNING)
    
    async def pause(self) -> bool:
        """暂停会话"""
        return await self.transition(SessionState.PAUSED)
    
    async def resume(self) -> bool:
        """恢复会话"""
        return await self.transition(SessionState.RUNNING)
    
    async def end(self) -> bool:
        """结束会话（进入结束流程，最终转为 COMPLETED 或 CANCELLED）"""
        return await self.transition(SessionState.ENDING)
```

## 六、服务层实现

```python
class SessionService:
    """会话服务实现"""
    
    def __init__(
        self,
        repository: ISessionRepository,
        manager: ISessionManager,
        event_bus: IEventBus,
        state_machine_factory: ISessionStateMachineFactory
    ):
        self.repository = repository
        self.manager = manager
        self.event_bus = event_bus
        self.state_machine_factory = state_machine_factory
        self._state_managers: Dict[str, SessionStateManager] = {}
    
    async def create_session(self, session_data: Dict[str, Any]) -> Session:
        """创建会话"""
        # 生成会话ID
        session_id = f"session_{uuid.uuid4().hex[:12]}"
        
        # 创建会话对象
        session = Session(
            id=session_id,
            topic=session_data.get("topic", ""),
            description=session_data.get("description", ""),
            state=SessionState.CREATED,
            config=SessionConfig.from_dict(session_data.get("config", {})),
            created_at=datetime.now(),
            updated_at=datetime.now()
        )
        
        # 保存到数据库
        session = await self.repository.create(session)
        
        # 创建状态管理器
        state_manager = self.state_machine_factory.create(session_id)
        self._state_managers[session_id] = state_manager
        
        # 添加参与者
        for participant_data in session_data.get("participants", []):
            participant = Participant(
                id=f"p_{uuid.uuid4().hex[:8]}",
                session_id=session_id,
                role_id=participant_data.get("role_id"),
                name=participant_data.get("name", ""),
                type=ParticipantType(participant_data.get("type", "ai")),
                created_at=datetime.now()
            )
            await self.manager.add_participant(session_id, participant)
        
        # 发布创建事件
        await self.event_bus.publish(create_session_event(
            event_type=EventType.SESSION_CREATED,
            session_id=session_id,
            session_data=session.to_dict()
        ))
        
        return session
    
    async def start_session(self, session_id: str) -> Session:
        """开始会话"""
        state_manager = self._state_managers.get(session_id)
        if not state_manager:
            raise SessionNotFoundError(session_id)
        
        # 初始化会话
        await state_manager.transition(SessionState.INITIALIZING)
        
        # 加载角色和规则
        # ... 初始化逻辑
        
        await state_manager.transition(SessionState.RUNNING)
        
        # 发布开始事件
        await self.event_bus.publish(create_session_event(
            event_type=EventType.SESSION_STARTED,
            session_id=session_id
        ))
        
        session = await self.repository.get_by_id(session_id)
        return session
    
    async def send_message(
        self,
        session_id: str,
        content: str,
        sender_id: str
    ) -> Message:
        """发送消息"""
        # 验证会话状态
        state_manager = self._state_managers.get(session_id)
        if not state_manager or state_manager.state != SessionState.RUNNING:
            raise SessionNotRunningError(session_id)
        
        # 创建消息
        message = Message(
            id=f"msg_{uuid.uuid4().hex[:12]}",
            session_id=session_id,
            sender_id=sender_id,
            content=content,
            timestamp=datetime.now()
        )
        
        # 保存消息
        message = await self.repository.save_message(message)
        
        # 发布消息接收事件
        await self.event_bus.publish(create_message_event(
            event_type=EventType.MESSAGE_RECEIVED,
            session_id=session_id,
            message_id=message.id,
            content=content,
            sender_id=sender_id
        ))
        
        return message
    
    async def end_session(self, session_id: str) -> Session:
        """结束会话"""
        state_manager = self._state_managers.get(session_id)
        if not state_manager:
            raise SessionNotFoundError(session_id)
        
        await state_manager.transition(SessionState.ENDING)
        
        # 清理资源
        # ... 清理逻辑
        
        # 清理完成后转为终态 COMPLETED
        await state_manager.transition(SessionState.COMPLETED)
        
        # 发布结束事件
        await self.event_bus.publish(create_session_event(
            event_type=EventType.SESSION_ENDED,
            session_id=session_id,
            session_data={"ended_at": datetime.now().isoformat()}
        ))
        
        # 清理状态管理器
        del self._state_managers[session_id]
        
        session = await self.repository.get_by_id(session_id)
        return session
```

## 七、会话管理器实现

```python
class SessionManager:
    """会话管理器实现"""
    
    def __init__(
        self,
        repository: ISessionRepository,
        event_bus: IEventBus
    ):
        self.repository = repository
        self.event_bus = event_bus
        self._contexts: Dict[str, SessionContext] = {}
        self._locks: Dict[str, asyncio.Lock] = defaultdict(asyncio.Lock)
    
    async def get_session_context(self, session_id: str) -> SessionContext:
        """获取会话上下文"""
        if session_id in self._contexts:
            return self._contexts[session_id]
        
        # 从数据库加载
        session = await self.repository.get_by_id(session_id)
        if not session:
            raise SessionNotFoundError(session_id)
        
        messages = await self.repository.get_messages(session_id)
        participants = await self._load_participants(session_id)
        
        context = SessionContext(
            session=session,
            participants=participants,
            messages=messages
        )
        
        self._contexts[session_id] = context
        return context
    
    async def add_participant(
        self,
        session_id: str,
        participant: Participant
    ) -> None:
        """添加参与者"""
        context = await self.get_session_context(session_id)
        context.add_participant(participant)
    
    async def remove_participant(
        self,
        session_id: str,
        participant_id: str
    ) -> None:
        """移除参与者"""
        context = await self.get_session_context(session_id)
        context.remove_participant(participant_id)
    
    async def broadcast_message(
        self,
        session_id: str,
        message: Message
    ) -> None:
        """广播消息"""
        context = await self.get_session_context(session_id)
        
        for participant in context.participants:
            if participant.id != message.sender_id:
                # 发布消息事件
                await self.event_bus.publish(create_event(
                    event_type=EventType.MESSAGE_DELIVERED,
                    source="session_manager",
                    session_id=session_id,
                    data={
                        "message_id": message.id,
                        "recipient_id": participant.id
                    }
                ))
    
    async def update_context(self, session_id: str) -> None:
        """更新会话上下文"""
        if session_id in self._contexts:
            session = await self.repository.get_by_id(session_id)
            messages = await self.repository.get_messages(session_id)
            
            self._contexts[session_id].session = session
            self._contexts[session_id].messages = messages
```

## 八、事件订阅

```python
class SessionEventHandler:
    """会话事件处理器"""
    
    def __init__(
        self,
        session_service: ISessionService,
        heartbeat_engine: IHeartbeatEngine
    ):
        self.session_service = session_service
        self.heartbeat_engine = heartbeat_engine
    
    @on_event(EventType.SESSION_STARTED)
    async def on_session_started(self, event: Event) -> None:
        """会话开始时启动心跳"""
        session_id = event.session_id
        await self.heartbeat_engine.start(session_id)
    
    @on_event(EventType.SESSION_PAUSED)
    async def on_session_paused(self, event: Event) -> None:
        """会话暂停时暂停心跳"""
        session_id = event.session_id
        await self.heartbeat_engine.pause(session_id)
    
    @on_event(EventType.SESSION_RESUMED)
    async def on_session_resumed(self, event: Event) -> None:
        """会话恢复时恢复心跳"""
        session_id = event.session_id
        await self.heartbeat_engine.resume(session_id)
    
    @on_event(EventType.SESSION_ENDED)
    async def on_session_ended(self, event: Event) -> None:
        """会话结束时停止心跳"""
        session_id = event.session_id
        await self.heartbeat_engine.stop(session_id)
```

## 九、数据库表结构

```sql
CREATE TABLE sessions (
    id VARCHAR(64) PRIMARY KEY,
    topic VARCHAR(512) NOT NULL DEFAULT '',
    description TEXT,
    state VARCHAR(32) NOT NULL DEFAULT 'created',
    config JSON NOT NULL DEFAULT '{}',
    metadata JSON NOT NULL DEFAULT '{}',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_state (state),
    INDEX idx_created_at (created_at)
);

CREATE TABLE participants (
    id VARCHAR(64) PRIMARY KEY,
    session_id VARCHAR(64) NOT NULL,
    role_id VARCHAR(64),
    name VARCHAR(128) NOT NULL,
    type VARCHAR(32) NOT NULL DEFAULT 'ai',
    metadata JSON NOT NULL DEFAULT '{}',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_session_id (session_id),
    FOREIGN KEY (session_id) REFERENCES sessions(id) ON DELETE CASCADE
);

CREATE TABLE messages (
    id VARCHAR(64) PRIMARY KEY,
    session_id VARCHAR(64) NOT NULL,
    sender_id VARCHAR(64) NOT NULL,
    content TEXT NOT NULL,
    type VARCHAR(32) NOT NULL DEFAULT 'text',
    metadata JSON NOT NULL DEFAULT '{}',
    timestamp TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_session_id (session_id),
    INDEX idx_sender_id (sender_id),
    INDEX idx_timestamp (timestamp),
    FOREIGN KEY (session_id) REFERENCES sessions(id) ON DELETE CASCADE
);
```

## 十、异常定义

```python
class SessionError(Exception):
    """会话基础异常"""
    pass

class SessionNotFoundError(SessionError):
    """会话不存在"""
    def __init__(self, session_id: str):
        self.session_id = session_id
        super().__init__(f"Session not found: {session_id}")

class SessionNotRunningError(SessionError):
    """会话未运行"""
    def __init__(self, session_id: str):
        self.session_id = session_id
        super().__init__(f"Session not running: {session_id}")

class InvalidStateTransitionError(SessionError):
    """无效的状态转换"""
    def __init__(self, from_state: str, to_state: str):
        self.from_state = from_state
        self.to_state = to_state
        super().__init__(f"Invalid state transition: {from_state} -> {to_state}")
```
