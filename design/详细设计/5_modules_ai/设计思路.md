# AI引擎模块设计

## 一、设计目标

AI引擎模块是系统的核心模块之一，负责：
1. **AI Agent管理**：管理会话中的AI角色代理
2. **决策引擎**：决定AI何时说话、说什么
3. **心理活动系统**：生成AI的心理活动描述
4. **情绪系统**：管理AI的情绪状态

## 二、设计原则

1. **角色一致性**：确保AI行为符合角色定义
2. **事件驱动**：AI行为由事件触发
3. **可扩展性**：支持新的AI行为模式
4. **上下文感知**：基于对话历史生成响应

## 五、AI Agent实现

```python
class AIAgent:
    """AI Agent实现"""
    
    def __init__(
        self,
        agent_id: str,
        role: Role,
        llm_adapter: ILLMAdapter,
        decision_engine: IAIDecisionEngine,
        mental_system: IMentalSystem,
        emotional_system: IEmotionalSystem,
        event_bus: IEventBus
    ):
        self.agent_id = agent_id
        self.role = role
        self.llm_adapter = llm_adapter
        self.decision_engine = decision_engine
        self.mental_system = mental_system
        self.emotional_system = emotional_system
        self.event_bus = event_bus
        
        # 状态
        self._message_count = 0
        self._last_speak_time = None
        self._is_thinking = False
    
    async def think(self, context: Dict[str, Any]) -> AIDecision:
        """思考过程"""
        self._is_thinking = True
        
        try:
            # 发布开始思考事件
            await self.event_bus.publish(create_ai_event(
                event_type=EventType.AI_THINKING,
                session_id=context.get("session_id"),
                agent_id=self.agent_id
            ))
            
            # 决策：是否发言
            decision = await self.decision_engine.should_speak(
                agent_id=self.agent_id,
                context=context
            )
            
            # 生成心理活动
            mental_activity = await self.mental_system.generate_mental_activity(
                agent_id=self.agent_id,
                context=context
            )
            
            # 更新心理状态
            if mental_activity:
                await self.mental_system.update_state(self.agent_id, mental_activity)
            
            # 发布思考完成事件
            await self.event_bus.publish(create_ai_event(
                event_type=EventType.AI_THINK,
                session_id=context.get("session_id"),
                agent_id=self.agent_id,
                mental_activity=mental_activity.content if mental_activity else None
            ))
            
            return decision
        
        finally:
            self._is_thinking = False
    
    async def act(self, decision: AIDecision) -> str:
        """行动：生成并返回响应"""
        if not decision.should_speak:
            return ""
        
        # 构建系统提示词
        system_prompt = await self._build_system_prompt()
        
        # 构建用户提示词
        user_prompt = self._build_user_prompt(decision.context)
        
        # 调用LLM
        response = await self.llm_adapter.chat(
            system_prompt=system_prompt,
            user_prompt=user_prompt,
            parameters=self.role.parameters
        )
        
        # 更新情绪（基于响应）
        await self._update_emotion_from_response(response)
        
        # 更新统计
        self._message_count += 1
        self._last_speak_time = datetime.now()
        
        # 发布行动事件
        await self.event_bus.publish(create_ai_event(
            event_type=EventType.AI_ACT,
            session_id=decision.context.get("session_id"),
            agent_id=self.agent_id,
            content=response
        ))
        
        return response
    
    async def _build_system_prompt(self) -> str:
        """构建系统提示词"""
        prompt_parts = []
        
        # 基础系统提示词
        if self.role.system_prompt:
            prompt_parts.append(self.role.system_prompt)
        
        # 角色性格描述
        personality_desc = self._describe_personality()
        prompt_parts.append(f"\n## 性格特征\n{personality_desc}")
        
        # 发言风格
        if self.role.speaking_style:
            style_desc = self._describe_speaking_style()
            prompt_parts.append(f"\n## 发言风格\n{style_desc}")
        
        return "\n\n".join(prompt_parts)
    
    def _describe_personality(self) -> str:
        """描述角色性格"""
        parts = []
        for trait, value in self.role.base_personality.items():
            if value > 0.7:
                parts.append(f"- {trait}: 非常{value}")
            elif value > 0.5:
                parts.append(f"- {trait}: 较为{value}")
            elif value < 0.3:
                parts.append(f"- {trait}: 很少{value}")
        return "\n".join(parts) if parts else "性格平衡"
    
    def _describe_speaking_style(self) -> str:
        """描述发言风格"""
        if not self.role.speaking_style:
            return "使用自然的对话风格"
        
        style = self.role.speaking_style
        parts = []
        
        if style.formality:
            parts.append(f"- 正式程度: {style.formality}")
        if style.length_preference:
            parts.append(f"- 长度偏好: {style.length_preference}")
        if style.tone:
            parts.append(f"- 语气: {style.tone}")
        
        return "\n".join(parts) if parts else "使用自然的对话风格"
    
    def _build_user_prompt(self, context: Dict[str, Any]) -> str:
        """构建用户提示词"""
        parts = []
        
        # 话题
        topic = context.get("topic", "")
        if topic:
            parts.append(f"## 当前话题\n{topic}")
        
        # 对话历史
        messages = context.get("messages", [])
        if messages:
            history = self._format_message_history(messages[-10:])  # 最近10条
            parts.append(f"\n## 对话历史\n{history}")
        
        # 当前情绪状态
        emotion = context.get("emotional_state")
        if emotion:
            parts.append(f"\n## 当前情绪\n情绪: {emotion.get('primary')}, 强度: {emotion.get('intensity')}")
        
        # 心理活动
        mental_activity = context.get("mental_activity")
        if mental_activity:
            parts.append(f"\n## 内心想法\n{mental_activity}")
        
        # 当前问题
        current_question = context.get("current_question")
        if current_question:
            parts.append(f"\n## 当前需要回应\n{current_question}")
        
        return "\n\n".join(parts)
    
    def _format_message_history(self, messages: List[Message]) -> str:
        """格式化消息历史"""
        lines = []
        for msg in messages:
            sender = msg.get("sender_name", msg.get("sender_id", "未知"))
            content = msg.get("content", "")
            lines.append(f"{sender}: {content}")
        return "\n".join(lines)
    
    async def _update_emotion_from_response(self, response: str) -> None:
        """从响应更新情绪"""
        # 简单的情绪分析
        positive_words = ["支持", "同意", "正确", "好", "赞"]
        negative_words = ["反对", "错误", "不好", "不对", "质疑"]
        
        positive_count = sum(1 for w in positive_words if w in response)
        negative_count = sum(1 for w in negative_words if w in response)
        
        if positive_count > negative_count:
            await self.emotional_system.update_emotion(
                agent_id=self.agent_id,
                trigger="positive_response",
                intensity=0.3
            )
        elif negative_count > positive_count:
            await self.emotional_system.update_emotion(
                agent_id=self.agent_id,
                trigger="negative_response",
                intensity=0.3
            )
```

```python
class AIDecisionEngine:
    """AI决策引擎"""
    
    def __init__(
        self,
        event_bus: IEventBus,
        config: DecisionConfig = None
    ):
        self.event_bus = event_bus
        self.config = config or DecisionConfig()
        self._agents: Dict[str, AIAgent] = {}
        self._cooldowns: Dict[str, datetime] = {}
    
    def register_agent(self, agent: AIAgent) -> None:
        """注册Agent"""
        self._agents[agent.agent_id] = agent
    
    async def should_speak(
        self,
        agent_id: str,
        context: Dict[str, Any]
    ) -> DecisionResult:
        """判断是否应该发言"""
        agent = self._agents.get(agent_id)
        if not agent:
            return DecisionResult(should_speak=False, reason="Agent not found")
        
        # 检查冷却时间
        if await self._is_in_cooldown(agent_id):
            return DecisionResult(should_speak=False, reason="Cooldown")
        
        # 检查心理配置
        mental_config = agent.role.mental_config
        if mental_config and mental_config.skip_probability:
            # 随机跳过
            if random.random() < mental_config.skip_probability:
                await self._publish_skip(agent_id, context, "Random skip")
                return DecisionResult(should_speak=False, reason="Random skip")
        
        # 检查消息频率
        msg_count = agent._message_count
        if msg_count > 0 and msg_count % 3 == 0:  # 每3条消息可能跳过
            if random.random() < 0.2:
                await self._publish_skip(agent_id, context, "Message frequency")
                return DecisionResult(should_speak=False, reason="Message frequency")
        
        # 检查上下文相关度
        topic = context.get("topic", "")
        relevance = await self.evaluate_topic(agent_id, topic)
        if relevance < 0.3:
            return DecisionResult(should_speak=False, reason="Low topic relevance")
        
        # 检查是否有需要回应的内容
        current_question = context.get("current_question")
        if not current_question:
            return DecisionResult(should_speak=False, reason="No question to respond")
        
        return DecisionResult(
            should_speak=True,
            reason="Normal response",
            context=context
        )
    
    async def generate_response(
        self,
        agent_id: str,
        context: Dict[str, Any],
        reasoning: Optional[str] = None
    ) -> AIResponse:
        """生成响应"""
        agent = self._agents.get(agent_id)
        if not agent:
            raise AgentNotFoundError(agent_id)
        
        # 思考
        decision = await agent.think(context)
        
        if not decision.should_speak:
            return AIResponse(
                content="",
                reasoning=decision.reason,
                mental_activity=None,
                skipped=True
            )
        
        # 行动
        content = await agent.act(decision)
        
        # 获取心理活动
        mental_activity = await self.mental_system.get_visible_activities(agent_id)
        
        # 设置冷却时间
        await self._set_cooldown(agent_id)
        
        return AIResponse(
            content=content,
            reasoning=reasoning or decision.reason,
            mental_activity=mental_activity,
            skipped=False
        )
    
    async def evaluate_topic(self, agent_id: str, topic: str) -> float:
        """评估话题相关度"""
        agent = self._agents.get(agent_id)
        if not agent:
            return 0.0
        
        # 简单实现：检查话题是否包含角色相关的关键词
        role_keywords = agent.role.variables.get("keywords", "").split(",")
        
        if not topic or not role_keywords:
            return 0.5
        
        topic_lower = topic.lower()
        matches = sum(1 for kw in role_keywords if kw.strip().lower() in topic_lower)
        
        return min(1.0, matches / max(len(role_keywords), 1))
    
    async def _is_in_cooldown(self, agent_id: str) -> bool:
        """检查是否在冷却中"""
        if agent_id not in self._cooldowns:
            return False
        
        cooldown_end = self._cooldowns[agent_id]
        if datetime.now() < cooldown_end:
            return True
        
        del self._cooldowns[agent_id]
        return False
    
    async def _set_cooldown(self, agent_id: str) -> None:
        """设置冷却时间"""
        interval = self.config.min_speak_interval
        self._cooldowns[agent_id] = datetime.now() + timedelta(seconds=interval)
    
    async def _publish_skip(
        self,
        agent_id: str,
        context: Dict[str, Any],
        reason: str
    ) -> None:
        """发布跳过事件"""
        await self.event_bus.publish(create_ai_event(
            event_type=EventType.AI_SKIP,
            session_id=context.get("session_id"),
            agent_id=agent_id,
            content=reason
        ))


@dataclass

    max_speak_interval: float = 30.0     # 最大发言间隔（秒）

class MentalSystem:
    """心理活动系统"""
    
    def __init__(
        self,
        llm_adapter: ILLMAdapter,
        event_bus: IEventBus,
        config: MentalConfig = None
    ):
        self.llm_adapter = llm_adapter
        self.event_bus = event_bus
        self.config = config or MentalConfig()
        self._activities: Dict[str, List[MentalActivity]] = {}
    
    async def generate_mental_activity(
        self,
        agent_id: str,
        context: Dict[str, Any]
    ) -> Optional[MentalActivity]:
        """生成心理活动"""
        if not self.config.enable_mental_activity:
            return None
        
        # 构建提示词
        prompt = self._build_mental_prompt(context)
        
        # 调用LLM生成心理活动
        try:
            content = await self.llm_adapter.chat(
                system_prompt="你是一个AI角色的内心独白生成器。请根据对话上下文，生成符合角色性格的内心想法。",
                user_prompt=prompt,
                parameters={"temperature": 0.8, "max_tokens": 200}
            )
            
            if not content:
                return None
            
            # 解析心理活动类型
            activity_type = self._classify_activity(content)
            
            # 创建心理活动
            activity = MentalActivity(
                id=f"mental_{uuid.uuid4().hex[:8]}",
                agent_id=agent_id,
                type=activity_type,
                content=content,
                timestamp=datetime.now(),
                visible=self._should_be_visible(activity_type)
            )
            
            # 存储
            if agent_id not in self._activities:
                self._activities[agent_id] = []
            self._activities[agent_id].append(activity)
            
            # 限制存储数量
            if len(self._activities[agent_id]) > 50:
                self._activities[agent_id] = self._activities[agent_id][-50:]
            
            return activity
        
        except Exception as e:
            logger.error(f"Failed to generate mental activity: {e}")
            return None
    
    def _build_mental_prompt(self, context: Dict[str, Any]) -> str:
        """构建心理活动提示词"""
        parts = []
        
        # 当前对话
        messages = context.get("messages", [])
        if messages:
            last_msg = messages[-1]
            parts.append(f"最近一条消息: {last_msg.get('content', '')}")
            parts.append(f"发送者: {last_msg.get('sender_name', '未知')}")
        
        # 当前情绪
        emotion = context.get("emotional_state")
        if emotion:
            parts.append(f"当前情绪: {emotion.get('primary')} (强度: {emotion.get('intensity')})")
        
        # 要求
        parts.append("\n请生成一段符合角色性格的内心独白，描述角色此刻的想法、感受或心理活动。")
        parts.append("格式：只需输出内心独白内容，不需要其他描述。")
        
        return "\n\n".join(parts)
    
    def _classify_activity(self, content: str) -> MentalActivityType:
        """分类心理活动类型"""
        content_lower = content.lower()
        
        if any(kw in content_lower for kw in ["思考", "想", "认为", "分析"]):
            return MentalActivityType.THINKING
        elif any(kw in content_lower for kw in ["感觉", "感受", "觉得", "情绪"]):
            return MentalActivityType.FEELING
        elif any(kw in content_lower for kw in ["计划", "打算", "准备", "下一步"]):
            return MentalActivityType.PLANNING
        elif any(kw in content_lower for kw in ["回忆", "想起", "记得", "过去"]):
            return MentalActivityType.RECALLING
        else:
            return MentalActivityType.IDLE
    
    def _should_be_visible(self, activity_type: MentalActivityType) -> bool:
        """判断心理活动是否应该可见"""
        visibility = self.config.mental_activity_visibility or 0.3
        
        # 不同类型的可见性
        type_visibility = {
            MentalActivityType.THINKING: 0.2,
            MentalActivityType.FEELING: 0.5,
            MentalActivityType.PLANNING: 0.1,
            MentalActivityType.RECALLING: 0.3,
            MentalActivityType.IDLE: 0.1
        }
        
        return random.random() < type_visibility.get(activity_type, 0.3) * visibility
    
    async def get_visible_activities(
        self,
        agent_id: str
    ) -> List[MentalActivity]:
        """获取可见的心理活动"""
        activities = self._activities.get(agent_id, [])
        return [a for a in activities if a.visible]

```python
class EmotionalSystem:
    """情绪系统"""
    
    def __init__(
        self,
        event_bus: IEventBus,
        config: EmotionalConfig = None
    ):
        self.event_bus = event_bus
        self.config = config or EmotionalConfig()
        self._states: Dict[str, EmotionalState] = {}
        self._history: Dict[str, List[EmotionalState]] = {}
    
    async def update_emotion(
        self,
        agent_id: str,
        trigger: str,
        intensity: float
    ) -> EmotionalState:
        """更新情绪"""
        current = self._states.get(agent_id)
        
        # 计算新情绪
        new_emotion = self._calculate_emotion(trigger, intensity, current)
        
        # 创建新状态
        state = EmotionalState(
            agent_id=agent_id,
            primary=new_emotion,
            intensity=min(1.0, intensity),
            trigger=trigger,
            timestamp=datetime.now()
        )
        
        # 存储状态
        self._states[agent_id] = state
        
        if agent_id not in self._history:
            self._history[agent_id] = []
        self._history[agent_id].append(state)
        
        # 限制历史长度
        if len(self._history[agent_id]) > 100:
            self._history[agent_id] = self._history[agent_id][-100:]
        
        return state
    
    def _calculate_emotion(
        self,
        trigger: str,
        intensity: float,
        current: Optional[EmotionalState]
    ) -> str:
        """计算情绪"""
        trigger_emotions = {
            "positive_response": "happy",
            "negative_response": "upset",
            "agreement": "satisfied",
            "disagreement": "frustrated",
            "question": "curious",
            "compliment": "pleased",
            "criticism": "angry"
        }
        
        emotion = trigger_emotions.get(trigger, "neutral")
        
        # 考虑当前情绪的惯性
        if current:
            if random.random() < self.config.emotion_inertia:
                return current.primary
        
        return emotion
    
    async def get_current_emotion(self, agent_id: str) -> EmotionalState:
        """获取当前情绪"""
        if agent_id in self._states:
            return self._states[agent_id]
        
        # 返回默认情绪
        return EmotionalState(
            agent_id=agent_id,
            primary="neutral",
            intensity=0.5,
            trigger="initial",
            timestamp=datetime.now()
        )
    
    async def apply_to_response(
        self,
        agent_id: str,
        response: str
    ) -> str:
        """将情绪影响应用到响应"""
        state = await self.get_current_emotion(agent_id)
        
        if state.intensity < 0.3:
            return response
        
        # 简单实现：根据情绪添加修饰
        emotion_prefix = {
            "happy": "（高兴地）",
            "upset": "（有些不悦）",
            "satisfied": "（满意地）",
            "frustrated": "（有些沮丧）",
            "curious": "（好奇地）",
            "pleased": "（愉快地）",
            "angry": "（生气地）",
            "neutral": ""
        }
        
        prefix = emotion_prefix.get(state.primary, "")
        if prefix and state.intensity > 0.5:
            return f"{prefix}{response}"
        
        return response
    
    async def decay_emotion(self, agent_id: str) -> None:
        """情绪衰减"""
        if agent_id not in self._states:
            return
        
        state = self._states[agent_id]
        new_intensity = state.intensity * (1 - self.config.decay_rate)
        
        if new_intensity < 0.1:
            # 恢复到中性
            self._states[agent_id] = EmotionalState(
                agent_id=agent_id,
                primary="neutral",
                intensity=0.5,
                trigger="decayed",
                timestamp=datetime.now()
            )
        else:
            state.intensity = new_intensity


@dataclass
class EmotionalConfig:
    """情绪配置"""
    emotion_inertia: float = 0.3      # 情绪惯性
    decay_rate: float = 0.1          # 衰减率
    max_intensity: float = 1.0       # 最大强度
    min_intensity: float = 0.0       # 最小强度
