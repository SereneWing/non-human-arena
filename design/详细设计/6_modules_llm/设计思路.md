# LLM适配器模块设计

## 一、设计目标

LLM适配器模块是系统的核心模块之一，负责：
1. **统一接口**：提供统一的LLM调用接口
2. **多LLM支持**：支持多种LLM后端（OpenAI、Anthropic、Ollama等）
3. **适配器模式**：使用适配器模式解耦业务逻辑和LLM实现
4. **配置管理**：管理不同LLM的配置和认证

## 二、设计原则

1. **接口抽象**：通过协议定义统一接口
2. **适配器模式**：每种LLM实现一个适配器
3. **配置分离**：LLM配置与业务代码分离
4. **错误处理**：统一的错误处理和重试机制

## 五、适配器基类实现

```python
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Optional, Dict, Any, AsyncIterator
import logging

logger = logging.getLogger(__name__)


@dataclass
class LLMConfig:
    """LLM配置基类"""
    model: str = "gpt-3.5-turbo"
    api_key: Optional[str] = None
    base_url: Optional[str] = None
    timeout: float = 60.0
    max_retries: int = 3
    temperature: float = 0.7
    max_tokens: int = 1000
    top_p: float = 0.9
    frequency_penalty: float = 0.0
    presence_penalty: float = 0.0


@dataclass
class ChatMessage:
    """聊天消息"""
    role: str  # system, user, assistant
    content: str


@dataclass
class ChatResponse:
    """聊天响应"""
    content: str
    model: str
    usage: Dict[str, int]  # prompt_tokens, completion_tokens, total_tokens
    finish_reason: str


class BaseLLMAdapter(ABC):
    """LLM适配器基类"""
    
    def __init__(self, config: LLMConfig):
        self.config = config
        self._validate_config()
    
    @property
    @abstractmethod
    def name(self) -> str:
        """适配器名称"""
        pass
    
    @property
    def model(self) -> str:
        """模型名称"""
        return self.config.model
    
    @abstractmethod
    async def _call_api(
        self,
        messages: List[ChatMessage],
        parameters: Dict[str, Any]
    ) -> ChatResponse:
        """调用API（子类实现）"""
        pass
    
    async def chat(
        self,
        system_prompt: str,
        user_prompt: str,
        parameters: Optional[Dict[str, Any]] = None
    ) -> str:
        """发送对话请求"""
        # 构建消息列表
        messages = [
            ChatMessage(role="system", content=system_prompt),
            ChatMessage(role="user", content=user_prompt)
        ]
        
        # 合并参数
        params = {**self._get_default_params(), **(parameters or {})}
        
        # 调用API
        response = await self._call_api(messages, params)
        
        return response.content
    
    async def chat_stream(
        self,
        system_prompt: str,
        user_prompt: str,
        parameters: Optional[Dict[str, Any]] = None
    ) -> AsyncIterator[str]:
        """流式对话请求"""
        messages = [
            ChatMessage(role="system", content=system_prompt),
            ChatMessage(role="user", content=user_prompt)
        ]
        
        params = {**self._get_default_params(), **(parameters or {})}
        params["stream"] = True
        
        async for chunk in self._call_stream_api(messages, params):
            yield chunk
    
    @abstractmethod
    async def _call_stream_api(
        self,
        messages: List[ChatMessage],
        parameters: Dict[str, Any]
    ) -> AsyncIterator[str]:
        """流式调用API（子类实现）"""
        pass
    
    async def count_tokens(self, text: str) -> int:
        """计算token数量（简单估算）"""
        # 简单估算：中文约2字符/token，英文约4字符/token
        chinese_chars = sum(1 for c in text if '\u4e00' <= c <= '\u9fff')
        other_chars = len(text) - chinese_chars
        return int(chinese_chars / 2 + other_chars / 4)
    
    def _get_default_params(self) -> Dict[str, Any]:
        """获取默认参数"""
        return {
            "temperature": self.config.temperature,
            "max_tokens": self.config.max_tokens,
            "top_p": self.config.top_p,
            "frequency_penalty": self.config.frequency_penalty,
            "presence_penalty": self.config.presence_penalty
        }
    
    def _validate_config(self) -> None:
        """验证配置"""
        if not self.config.model:
            raise ValueError("Model name is required")
    
    async def validate_config(self) -> bool:
        """验证配置是否有效"""
        try:
            self._validate_config()
            return True
        except Exception as e:
            logger.error(f"LLM config validation failed: {e}")
            return False
```

## 六、OpenAI适配器实现

```python
import httpx
from typing import AsyncIterator
import json

class OpenAIAdapter(BaseLLMAdapter):
    """OpenAI适配器"""
    
    def __init__(self, config: LLMConfig):
        super().__init__(config)
        self.base_url = config.base_url or "https://api.openai.com/v1"
        self.api_key = config.api_key
    
    @property
    def name(self) -> str:
        return "openai"
    
    async def _call_api(
        self,
        messages: List[ChatMessage],
        parameters: Dict[str, Any]
    ) -> ChatResponse:
        """调用OpenAI API"""
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        
        payload = {
            "model": self.config.model,
            "messages": [{"role": m.role, "content": m.content} for m in messages],
            **parameters
        }
        
        async with httpx.AsyncClient(timeout=self.config.timeout) as client:
            response = await client.post(
                f"{self.base_url}/chat/completions",
                headers=headers,
                json=payload
            )
            
            if response.status_code != 200:
                raise LLMAPIError(
                    f"OpenAI API error: {response.status_code}",
                    status_code=response.status_code,
                    response=response.text
                )
            
            data = response.json()
            
            return ChatResponse(
                content=data["choices"][0]["message"]["content"],
                model=data.get("model", self.config.model),
                usage=data.get("usage", {}),
                finish_reason=data["choices"][0].get("finish_reason", "stop")
            )
    
    async def _call_stream_api(
        self,
        messages: List[ChatMessage],
        parameters: Dict[str, Any]
    ) -> AsyncIterator[str]:
        """流式调用OpenAI API"""
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        
        payload = {
            "model": self.config.model,
            "messages": [{"role": m.role, "content": m.content} for m in messages],
            **parameters,
            "stream": True
        }
        
        async with httpx.AsyncClient(timeout=self.config.timeout) as client:
            async with client.stream(
                "POST",
                f"{self.base_url}/chat/completions",
                headers=headers,
                json=payload
            ) as response:
                async for line in response.aiter_lines():
                    if line.startswith("data: "):
                        data_str = line[6:]
                        if data_str == "[DONE]":
                            break
                        
                        data = json.loads(data_str)
                        if "choices" in data and len(data["choices"]) > 0:
                            delta = data["choices"][0].get("delta", {})
                            if "content" in delta:
                                yield delta["content"]


class LLMAPIError(Exception):
    """LLM API错误"""
    
    def __init__(
        self,
        message: str,
        status_code: int = None,
        response: str = None
    ):
        super().__init__(message)
        self.status_code = status_code
        self.response = response
```

## 七、LLM管理器实现

```python
class LLMManager:
    """LLM管理器"""
    
    def __init__(self, event_bus: IEventBus = None):
        self.event_bus = event_bus
        self._adapters: Dict[str, ILLMAdapter] = {}
        self._default_adapter: str = None
        self._lock = asyncio.Lock()
    
    async def get_adapter(self, adapter_name: str = None) -> ILLMAdapter:
        """获取适配器"""
        name = adapter_name or self._default_adapter
        
        if not name:
            raise ValueError("No adapter name specified and no default adapter set")
        
        if name not in self._adapters:
            raise ValueError(f"Adapter not found: {name}")
        
        return self._adapters[name]
    
    async def register_adapter(
        self,
        name: str,
        adapter: ILLMAdapter
    ) -> None:
        """注册适配器"""
        async with self._lock:
            self._adapters[name] = adapter
            
            if self._default_adapter is None:
                self._default_adapter = name
        
        logger.info(f"Registered LLM adapter: {name}")
    
    async def list_adapters(self) -> List[str]:
        """列出所有适配器"""
        return list(self._adapters.keys())
    
    async def set_default(self, adapter_name: str) -> None:
        """设置默认适配器"""
        if adapter_name not in self._adapters:
            raise ValueError(f"Adapter not found: {adapter_name}")
        
        self._default_adapter = adapter_name
    
    async def create_adapter(
        self,
        adapter_type: str,
        config: Dict[str, Any]
    ) -> ILLMAdapter:
        """创建适配器"""
        # 创建配置
        llm_config = LLMConfig(
            model=config.get("model", "gpt-3.5-turbo"),
            api_key=config.get("api_key"),
            base_url=config.get("base_url"),
            timeout=config.get("timeout", 60.0),
            temperature=config.get("temperature", 0.7),
            max_tokens=config.get("max_tokens", 1000)
        )
        
        # 根据类型创建适配器
        if adapter_type == "openai":
            adapter = OpenAIAdapter(llm_config)
        elif adapter_type == "anthropic":
            adapter = AnthropicAdapter(llm_config)
        elif adapter_type == "ollama":
            adapter = OllamaAdapter(llm_config)
        else:
            raise ValueError(f"Unknown adapter type: {adapter_type}")
        
        # 注册
        await self.register_adapter(adapter_type, adapter)
        
        return adapter
    
    async def remove_adapter(self, name: str) -> None:
        """移除适配器"""
        async with self._lock:
            if name in self._adapters:
                del self._adapters[name]
                
                if self._default_adapter == name:
                    self._default_adapter = next(iter(self._adapters), None)
    
    async def get_default_adapter_name(self) -> Optional[str]:
        """获取默认适配器名称"""
        return self._default_adapter
```

## 八、工厂模式

```python
class LLMAdapterFactory:
    """LLM适配器工厂"""
    
    _registry: Dict[str, type] = {}
    
    @classmethod
    def register(cls, adapter_type: str, adapter_class: type) -> None:
        """注册适配器类"""
        cls._registry[adapter_type] = adapter_class
    
    @classmethod
    def create(cls, adapter_type: str, config: LLMConfig) -> ILLMAdapter:
        """创建适配器"""
        if adapter_type not in cls._registry:
            raise ValueError(f"Unknown adapter type: {adapter_type}")
        
        adapter_class = cls._registry[adapter_type]
        return adapter_class(config)
    
    @classmethod
    def list_types(cls) -> List[str]:
        """列出支持的类型"""
        return list(cls._registry.keys())


# 注册内置适配器
LLMAdapterFactory.register("openai", OpenAIAdapter)
LLMAdapterFactory.register("anthropic", AnthropicAdapter)
LLMAdapterFactory.register("ollama", OllamaAdapter)
```

## 九、使用示例

```python
# 方式1：直接创建适配器
config = LLMConfig(
    model="gpt-4",
    api_key="sk-xxx"
)
adapter = OpenAIAdapter(config)
response = await adapter.chat(
    system_prompt="你是一个有帮助的助手",
    user_prompt="你好"
)

# 方式2：使用管理器
manager = LLMManager()

# 注册适配器
await manager.create_adapter("openai", {
    "model": "gpt-4",
    "api_key": "sk-xxx"
})

await manager.create_adapter("ollama", {
    "model": "llama2",
    "base_url": "http://localhost:11434"
})

# 使用默认适配器
adapter = await manager.get_adapter()
response = await adapter.chat(system_prompt, user_prompt)

# 使用指定适配器
adapter = await manager.get_adapter("ollama")
response = await adapter.chat(system_prompt, user_prompt)

# 流式响应
async for chunk in adapter.chat_stream(system_prompt, user_prompt):
    print(chunk, end="", flush=True)
```
